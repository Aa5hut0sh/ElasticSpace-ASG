#!/bin/bash

cd /home/ubuntu/agent

# load nvm
export NVM_DIR="/home/ubuntu/.nvm"
source $NVM_DIR/nvm.sh

# ensure node in path
export PATH=$PATH:/home/ubuntu/.nvm/versions/node/v24.13.0/bin/

# install dependencies
npm install

# install pm2
npm install -g pm2

# start agent
pm2 start agent.js --name agent

# persist pm2
pm2 startup systemd -u ubuntu --hp /home/ubuntu
pm2 save

# run coder container
docker run -d \
  -p 8080:8080 \
  --restart=always \
  --name coder \
  codercom
